12uint32_t mass_color(double m) {
13 const double t = std::min(m / 5.0, 1.0);
14 uint8_t r_, g_, b_;
15 if (t < 0.5) {
16 float u = float(t * 2.0);
17 r_ = uint8_t(120 + u * 135); // 120..255
18 g_ = uint8_t(200 + u * 38); // 200..238
19 b_ = uint8_t(255 - u * 85); // 255..170
20 } else {
21 float u = float((t - 0.5) * 2.0);
22 r_ = 255;
23 g_ = uint8_t(238 - u * 136); // 238..102
24 b_ = uint8_t(170 - u * 136); // 170..34
25 }
26 return (uint32_t(r_) << 24) | (uint32_t(g_) << 16) | (uint32_t(b_) << 8)
27 | 0xFFu;
28}

40void NBodySim::make_accel(const num::Vector& pos, num::Vector& acc) const {
41 const int nb = n();
42 std::fill(acc.begin(), acc.end(), 0.0);
43 num::Vec2ConstView p{pos};
44 num::Vec2View a{acc};
45 for (int i = 0; i < nb; ++i) {
46 for (int j = 0; j < nb; ++j) {
47 if (i == j)
48 continue;
49 double dx = p.x(j) - p.x(i);
50 double dy = p.y(j) - p.y(i);
51 double r2 = dx * dx + dy * dy + soft * soft;
52 double fac = G * bodies[j].mass / (r2 * std::sqrt(r2));
53 a.x(i) += fac * dx;
54 a.y(i) += fac * dy;
55 }
56 }
57}

59double NBodySim::kinetic_energy() const {
60 num::Vec2ConstView vel{v};
61 double KE = 0.0;
62 for (int i = 0; i < n(); ++i)
63 KE += 0.5 * bodies[i].mass
64 * (vel.x(i) * vel.x(i) + vel.y(i) * vel.y(i));
65 return KE;
66}

68double NBodySim::potential_energy() const {
69 num::Vec2ConstView pos{q};
70 double PE = 0.0;
71 for (int i = 0; i < n(); ++i) {
72 for (int j = i + 1; j < n(); ++j) {
73 double dx = pos.x(j) - pos.x(i);
74 double dy = pos.y(j) - pos.y(i);
75 PE -= G * bodies[i].mass * bodies[j].mass
76 / std::sqrt(dx * dx + dy * dy + soft * soft);
77 }
78 }
79 return PE;
80}

82void NBodySim::step(double dt) {
83 auto accel = [this](const num::Vector& pos, num::Vector& acc) {
84 make_accel(pos, acc);
85 };
86 num::ODEParams p = {.t0 = t, .tf = t + dt, .h = dt};
87 num::SymplecticResult res;
88 switch (integrator) {
89 case Integrator::Verlet:
90 res = num::verlet(accel, q, v, p).run();
91 break;
92 case Integrator::Yoshida4:
93 res = num::yoshida4(accel, q, v, p).run();
94 break;
95 case Integrator::RK4:
96 res = num::rk4_2nd(accel, q, v, p).run();
97 break;
98 }
99 q = std::move(res.q);
100 v = std::move(res.v);
101 t += dt;
102}

104std::vector<std::pair<int, int>> NBodySim::check_merges() {
105 std::vector<std::pair<int, int>> ops;
106 int i = 0;
107 while (i < n() && n() > 1) {
108 bool merged = false;
109 num::Vec2View pos{q}, vel{v};
110 for (int j = i + 1; j < n(); ++j) {
111 double dx = pos.x(i) - pos.x(j);
112 double dy = pos.y(i) - pos.y(j);
113 double mrd = bodies[i].phys_radius + bodies[j].phys_radius;
114 if (dx * dx + dy * dy >= mrd * mrd)
115 continue;
116
117 // Merge j into i -- conserve mass and momentum
118 double mi = bodies[i].mass, mj = bodies[j].mass, mt = mi + mj;
119 pos.x(i) = (mi * pos.x(i) + mj * pos.x(j)) / mt;
120 pos.y(i) = (mi * pos.y(i) + mj * pos.y(j)) / mt;
121 vel.x(i) = (mi * vel.x(i) + mj * vel.x(j)) / mt;
122 vel.y(i) = (mi * vel.y(i) + mj * vel.y(j)) / mt;
123 bodies[i].mass = mt;
124 bodies[i].display_radius = disp_r(mt);
125 bodies[i].phys_radius = phys_r(mt);
126 bodies[i].color = mass_color(mt);
127
128 // Swap j with the last body and shrink arrays
129 int last = n() - 1;
130 ops.push_back({j, last});
131 if (j != last) {
132 std::swap(bodies[j], bodies[last]);
133 pos.x(j) = pos.x(last);
134 pos.y(j) = pos.y(last);
135 vel.x(j) = vel.x(last);
136 vel.y(j) = vel.y(last);
137 }
138 bodies.pop_back();
139
140 // Rebuild q and v at the new (smaller) size
141 int nn = n();
142 num::Vector nq(2 * nn, 0.0), nv(2 * nn, 0.0);
143 for (int k = 0; k < 2 * nn; ++k) {
144 nq[k] = q[k];
145 nv[k] = v[k];
146 }
147 q = std::move(nq);
148 v = std::move(nv);
149
150 merged = true;
151 break; // restart j-scan for body i (it may now touch another body)
152 }
153 if (!merged)
154 ++i;
155 }
156 return ops;
157}

159void NBodySim::reset(Scenario s) {
160 t = 0.0;
161 G = 1.0;
162 enable_merges = false;
163
164 if (s == Scenario::Figure8) {
165 soft = 1e-4;
166 bodies = {
167 {1.0, 10.0f, 0xFF5555FF},
168 {1.0, 10.0f, 0x55FF55FF},
169 {1.0, 10.0f, 0x5599FFFF},
170 };
171 q = {-0.97000436, 0.24308753, 0.97000436, -0.24308753, 0.0, 0.0};
172 v = {0.46620368,
173 0.43236573,
174 0.46620368,
175 0.43236573,
176 -0.93240737,
177 -0.86473146};
178
179 } else if (s == Scenario::SolarSystem) {
180 soft = 1e-3;
181 const double Msun = 1000.0;
182 bodies = {
183 {Msun, 18.0f, 0xFFDD33FF},
184 {1.0, 4.0f, 0xAAAAAAFF},
185 {1.0, 6.0f, 0x44AAFFFF},
186 {1.0, 5.0f, 0xFF6644FF},
187 {10.0, 11.0f, 0xFFAA44FF},
188 };
189 const double radii[] = {0.0, 0.22, 0.40, 0.60, 1.10};
190 const int nb = 5;
191 q = num::Vector(2 * nb, 0.0);
192 v = num::Vector(2 * nb, 0.0);
193 num::Vec2View pos{q}, vel{v};
194 for (int i = 1; i < nb; ++i) {
195 pos.x(i) = radii[i];
196 vel.y(i) = std::sqrt(G * Msun / radii[i]);
197 }
198
199 } else if (s == Scenario::BinaryPlus) {
200 soft = 1e-3;
201 const double Mstar = 50.0;
202 bodies = {
203 {Mstar, 13.0f, 0xFF7744FF},
204 {Mstar, 13.0f, 0x44AAFFFF},
205 {0.01, 5.0f, 0xAAFF88FF},
206 };
207 const double sep = 0.3;
208 const double omega = std::sqrt(G * 2.0 * Mstar / (sep * sep * sep));
209 double vs = omega * sep * 0.5;
210 double r_tp = 1.5;
211 double v_tp = std::sqrt(G * 2.0 * Mstar / r_tp) * 0.85;
212 q = {-sep * 0.5, 0.0, sep * 0.5, 0.0, r_tp, 0.0};
213 v = {0.0, -vs, 0.0, vs, 0.0, v_tp};
214
215 } else { // Galaxy
216 enable_merges = true;
217 soft = 0.10;
218
219 const int N_gal = 200;
220 const double R_disk = 3.0; // initial spread radius
221
222 bodies.clear();
223 q = num::Vector(2 * N_gal, 0.0);
224 v = num::Vector(2 * N_gal, 0.0);
225
226 std::mt19937 rng(20240101);
227 std::uniform_real_distribution<double> uni(0.0, 1.0);
228
229 // Estimate total mass to seed tangential speeds
230 // With masses uniform in [0.3, 2.0], mean ~= 1.15, M_total ~= 230
231 const double M_total_est = N_gal * 1.15;
232
233 for (int i = 0; i < N_gal; ++i) {
234 // Radius: slightly concentrated toward center
235 double r = R_disk * std::pow(uni(rng), 0.65);
236 double theta = uni(rng) * 2.0 * M_PI;
237
238 // Mass: random uniform [0.3, 2.0]
239 double m = 0.3 + 1.7 * uni(rng);
240
241 // Tangential velocity: ~35% of circular velocity at this radius
242 // This gives highly elliptical orbits that cross each other
243 double vcirc = (r > 0.05) ? std::sqrt(G * M_total_est / r) : 0.0;
244 double vtang = vcirc
245 * (0.25 + 0.20 * uni(rng)); // 25-45% of circular
246
247 // Small random perturbation so orbits have different phases
248 double vrand_x = (2.0 * uni(rng) - 1.0) * 0.3;
249 double vrand_y = (2.0 * uni(rng) - 1.0) * 0.3;
250
251 num::Vec2View pos{q}, vel{v};
252 pos.x(i) = r * std::cos(theta);
253 pos.y(i) = r * std::sin(theta);
254 // Tangential direction (counterclockwise): (-sin(theta),
255 // cos(theta))
256 vel.x(i) = -vtang * std::sin(theta) + vrand_x;
257 vel.y(i) = vtang * std::cos(theta) + vrand_y;
258
259 bodies.push_back({m, disp_r(m), mass_color(m), phys_r(m)});
260 }
261 }
262
263 E0 = total_energy();
264}
