23 , up(N * N)
24 , dn(N * N)
25 , lt(N * N)
26 , rt(N * N) {

28 for (int col = 0; col < N; ++col) {
29 const int i = row * N + col;
30 up[i] = ((row - 1 + N) % N) * N + col;
31 dn[i] = ((row + 1) % N) * N + col;
32 lt[i] = row * N + (col - 1 + N) % N;
33 rt[i] = row * N + (col + 1) % N;
34 }
