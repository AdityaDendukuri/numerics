18void heat_compute(std::vector<Particle3D>& particles,
19 const std::vector<RigidBody3D>& bodies,
20 const SpatialHash3D& grid,
21 const HeatParams3D& params) {
22#ifdef NUMERICS_HAS_OMP
23 const float h = params.h;
24 const float alpha = params.alpha_T;
25 const float m = params.mass;
26 const float h_conv = params.h_conv;
27 const float supp_sq = 4.0f * h * h;
28 const float eps2 = 0.01f * h * h;
29 const int n = static_cast<int>(particles.size());
30
31 #pragma omp parallel for schedule(static)
32 for (int i = 0; i < n; ++i) {
33 Particle3D& pi = particles[i];
34 float dT = 0.0f;
35
36 grid.query(pi.x, pi.y, pi.z, [&](int j) {
37 if (j == i)
38 return;
39 const Particle3D& pj = particles[j];
40 const float rx = pi.x - pj.x, ry = pi.y - pj.y, rz = pi.z - pj.z;
41 const float r2 = rx * rx + ry * ry + rz * rz;
42 if (r2 >= supp_sq || r2 < 1e-10f)
43 return;
44 const float r = std::sqrt(r2);
45 const float rij_dot_gradW = K::dW_dr(r, h) * r;
46 dT += 2.0f * alpha * (m / pj.density)
47 * (pi.temperature - pj.temperature) * rij_dot_gradW
48 / (r2 + eps2);
49 });
50
51 for (const RigidBody3D& body : bodies) {
52 const float dx = pi.x - body.x, dy = pi.y - body.y,
53 dz = pi.z - body.z;
54 const float d = std::sqrt(dx * dx + dy * dy + dz * dz);
55 if (d < body.radius + 2.0f * h) {
56 const float excess = std::max(0.0f, d - body.radius);
57 const float phi = 1.0f - excess / (2.0f * h);
58 dT += h_conv * (body.temperature - pi.temperature) * phi;
59 }
60 }
61
62 pi.dT_dt = dT;
63 }
64#else
65 seq::heat_compute(particles, bodies, grid, params);
66#endif
67}
