44// Hose emitter
45struct Hose {
46 Vector3 source, dir, perp_u, perp_v;
47 float temperature, speed, cone_half;
48
49 static Vector3 norm(Vector3 v) {
50 float len = sqrtf(v.x * v.x + v.y * v.y + v.z * v.z);
51 return len < 1e-8f ? Vector3{0, 1, 0}
52 : Vector3{v.x / len, v.y / len, v.z / len};
53 }
54 static Vector3 cross(Vector3 a, Vector3 b) {
55 return {a.y * b.z - a.z * b.y,
56 a.z * b.x - a.x * b.z,
57 a.x * b.y - a.y * b.x};
58 }

60 void init(Vector3 src,
61 Vector3 target,
62 float temp,
63 float spd,
64 float half_angle) {
65 source = src;
66 temperature = temp;
67 speed = spd;
68 cone_half = half_angle;
69 dir = norm({target.x - src.x, target.y - src.y, target.z - src.z});
70 Vector3 ref = fabsf(dir.y) < 0.9f ? Vector3{0, 1, 0} : Vector3{1, 0, 0};
71 perp_u = norm(cross(dir, ref));
72 perp_v = cross(dir, perp_u);
73 }

75 void emit(physics::FluidSolver3D& solver,
76 int count,
77 int max_n = 1800) const {
78 if (cast<int>(solver.particles().size()) >= max_n)
79 return;
80 const float max_r = tanf(cone_half);
81 for (int i = 0; i < count; ++i) {
82 float angle = cast<float>(GetRandomValue(0, 10000)) / 10000.0f
83 * 2.0f * kPI;
84 float r = cast<float>(GetRandomValue(0, 10000)) / 10000.0f * max_r;
85 float su = cosf(angle) * r, sv = sinf(angle) * r;
86 Vector3 d = norm({dir.x + su * perp_u.x + sv * perp_v.x,
87 dir.y + su * perp_u.y + sv * perp_v.y,
88 dir.z + su * perp_u.z + sv * perp_v.z});
89 float jx = cast<float>(GetRandomValue(-100, 100)) * 0.00008f;
90 float jy = cast<float>(GetRandomValue(-100, 100)) * 0.00008f;
91 float jz = cast<float>(GetRandomValue(-100, 100)) * 0.00008f;
92 solver.add_particle(source.x + jx,
93 source.y + jy,
94 source.z + jz,
95 d.x * speed,
96 d.y * speed,
97 d.z * speed,
98 temperature);
99 }
100 }
101};
