117void apply_1q(Statevector& sv, int n_qubits, const Gate& G, int target) {
118 const int stride = 1 << target;
119 const int N = 1 << n_qubits;
120 for (int i = 0; i < N; i += stride << 1) {
121 for (int j = 0; j < stride; ++j) {
122 const int i0 = i + j, i1 = i0 + stride;
123 const cplx a0 = sv[i0], a1 = sv[i1];
124 sv[i0] = G[0] * a0 + G[1] * a1;
125 sv[i1] = G[2] * a0 + G[3] * a1;
126 }
127 }
128}

130// Apply a 4x4 gate to qubits q0 (LSB) and q1 (MSB) of the 2-qubit subspace.
131// The gate acts on the 4D subspace spanned by |q1 q0> in {00,01,10,11}.
132void apply_2q(Statevector& sv, int n_qubits, const Gate2Q& G, int q0, int q1) {
133 const int N = 1 << n_qubits;
134 const int m0 = 1 << q0, m1 = 1 << q1;
135
136 for (int i = 0; i < N; ++i) {
137 // Only process each group of 4 once: skip unless both bits are 0
138 if (i & m0 || i & m1)
139 continue;
140 const int i00 = i, i01 = i | m0, i10 = i | m1, i11 = i | m0 | m1;
141 const cplx a00 = sv[i00], a01 = sv[i01], a10 = sv[i10], a11 = sv[i11];
142 sv[i00] = G[0] * a00 + G[1] * a01 + G[2] * a10 + G[3] * a11;
143 sv[i01] = G[4] * a00 + G[5] * a01 + G[6] * a10 + G[7] * a11;
144 sv[i10] = G[8] * a00 + G[9] * a01 + G[10] * a10 + G[11] * a11;
145 sv[i11] = G[12] * a00 + G[13] * a01 + G[14] * a10 + G[15] * a11;
146 }
147}

243std::array<cplx, 4> reduced_density_matrix(const Statevector& sv,
244 int n_qubits,
245 int qubit) {
246 // Partial trace over all qubits except `qubit`.
247 // rho_q[a,b] = Sigma_{env} alpha_{env|a} * conj(alpha_{env|b})
248 // where env iterates over the N/2 basis states with qubit bit = 0.
249 // This is O(N) -- not O(N^2).
250 const int N = 1 << n_qubits;
251 const int mask = 1 << qubit;
252 std::array<cplx, 4> rho{};
253 for (int i = 0; i < N; ++i) {
254 if (i & mask)
255 continue; // visit each environment state once
256 const int i1 = i | mask; // same env, qubit = 1
257 rho[0] += sv[i] * std::conj(sv[i]); // rho[0,0]
258 rho[1] += sv[i] * std::conj(sv[i1]); // rho[0,1]
259 rho[2] += sv[i1] * std::conj(sv[i]); // rho[1,0]
260 rho[3] += sv[i1] * std::conj(sv[i1]); // rho[1,1]
261 }
262 return rho;
263}

265real entanglement_entropy(const Statevector& sv, int n_qubits, int qubit) {
266 auto rho = reduced_density_matrix(sv, n_qubits, qubit);
267 // Eigenvalues of 2x2 Hermitian: lambda = (tr +/- sqrt(tr^2-4det)) / 2
268 real tr = rho[0].real() + rho[3].real();
269 real det = (rho[0] * rho[3] - rho[1] * rho[2]).real();
270 real disc = std::max(real(0), tr * tr / 4 - det);
271 real sq = std::sqrt(disc);
272 real l0 = tr / 2 + sq, l1 = tr / 2 - sq;
273 auto xlogx = [](real x) -> real {
274 return (x > 1e-15) ? -x * std::log2(x) : real(0);
275 };
276 return xlogx(l0) + xlogx(l1);
277}

279void apply_cy(Statevector& sv, int n_qubits, int control, int target) {
280 // CY = controlled-Y: apply Y to target when control=1
281 // Y = [[0,-i],[i,0]], so swap amplitudes with phase factor
282 const int N = 1 << n_qubits;
283 const int ctrl_mask = 1 << control, tgt_mask = 1 << target;
284 for (int i = 0; i < N; ++i) {
285 if ((i & ctrl_mask) && !(i & tgt_mask)) {
286 int j = i | tgt_mask;
287 cplx a0 = sv[i], a1 = sv[j];
288 sv[i] = cplx{0, 1} * a1; // i*a1
289 sv[j] = cplx{0, -1} * a0; // -i*a0
290 }
291 }
292}

294void apply_cp(Statevector& sv,
295 int n_qubits,
296 real lambda,
297 int control,
298 int target) {
299 // Controlled phase: |11> -> e^{ilambda}|11>, others unchanged
300 const int N = 1 << n_qubits;
301 const int ctrl_mask = 1 << control, tgt_mask = 1 << target;
302 const cplx phase{std::cos(lambda), std::sin(lambda)};
303 for (int i = 0; i < N; ++i) {
304 if ((i & ctrl_mask) && (i & tgt_mask))
305 sv[i] *= phase;
306 }
307}
