1/// @file include/ising/sim.hpp
2/// @brief 2D Ising model on an NxN periodic lattice.
3///
4/// Wraps the numerics library's Metropolis and umbrella-sampling sweeps into a
5/// self-contained IsingLattice. No rendering code -- pure physics.
6///
7/// Two sweep modes:
8/// sweep() -- classic Metropolis (free evolution)
9/// sweep_umbrella() -- umbrella sampling constrained to a nucleus-size
10/// window

12/// Observable accessors:
13/// magnetization() -- |<m>| via SIMD dot product

86 /// Square spin-down nucleus of area ~r centred at the middle of the
87 /// lattice.
88 void generate_nucleus(real r = 100.0) {
89 all_up();
90 int half = static_cast<int>(std::sqrt(r) * 0.5);
91 int cx = N / 2;
92 for (int row = cx - half; row < cx + half; ++row)
93 for (int col = cx - half; col < cx + half; ++col)
94 spins[row * N + col] = -1.0;
95 }

97 // Sweeps
98
99 /// Standard Metropolis sweep: N^2 random spin picks with replacement.
100 /// Uses a precomputed Boltzmann table -- no exp() at runtime.
101 void sweep() {
102 num::markov::metropolis_sweep_prob(
103 static_cast<idx>(NN),
104 [&](idx i) {
105 real s = spins[i];
106 real ns = spins[nbr_.up[i]] + spins[nbr_.dn[i]]
107 + spins[nbr_.lt[i]] + spins[nbr_.rt[i]];
108 return boltz_[spin_idx(s)][nbr_idx(ns)];
109 },
110 [&](idx i) { spins[i] = -spins[i]; },
111 rng_);
112 }
113
114 /// Umbrella sweep: revert if nucleus size leaves [lo, hi].
115 /// Fills det with the connected-component result for rendering.
116 int sweep_umbrella(int lo, int hi, num::ClusterResult& det) {
117 auto result = num::markov::umbrella_sweep_prob(
118 static_cast<idx>(NN),
119 [&](idx i) {
120 real s = spins[i];
121 real ns = spins[nbr_.up[i]] + spins[nbr_.dn[i]]
122 + spins[nbr_.lt[i]] + spins[nbr_.rt[i]];
123 return boltz_[spin_idx(s)][nbr_idx(ns)];
124 },
125 [&](idx i) { spins[i] = -spins[i]; },
126 [&]() {
127 for (int i = 0; i < NN; ++i)
128 save_buf_[i] = spins[i];
129 },
130 [&]() {
131 for (int i = 0; i < NN; ++i)
132 spins[i] = save_buf_[i];
133 },
134 [&]() -> idx {
135 det = num::connected_components(
136 NN,
137 [&](int i) { return spins[i] < 0.0; },
138 [&](int i, auto&& visit) {
139 visit(nbr_.up[i]);
140 visit(nbr_.dn[i]);
141 visit(nbr_.lt[i]);
142 visit(nbr_.rt[i]);
143 });
144 return static_cast<idx>(det.largest_size);
145 },
146 num::markov::UmbrellaWindow{static_cast<idx>(lo),
147 static_cast<idx>(hi)},
148 rng_);
149 return static_cast<int>(result.order_param);
150 }

152 // Observables
153
154 /// |<m>| = |sum(s_i)| / N^2 via SIMD dot product.
155 real magnetization() const {
156 return std::abs(dot(spins, ones_, num::Backend::seq))

158 }
159
160 /// <E>/N^2 using the sparse adjacency matrix and SIMD sparse_matvec.
161 real energy_per_spin() {
162 num::sparse_matvec(adj_, spins, nbr_buf_);
163 real coup = -J * dot(spins, nbr_buf_, num::Backend::seq) / (4.0 * NN);
164 real fld = -F * dot(spins, ones_, num::Backend::seq)
165 / static_cast<real>(NN);
166 return coup + fld;
167 }

169 /// Mean-field order parameter: solve m = tanh(4*beta*J*m + beta*F) via
170 /// num::newton().
171 static real mean_field_m(real beta_val, real F_val) {
172 auto f = [=](real m) {
173 return std::tanh(4.0 * beta_val * J * m + beta_val * F_val) - m;
174 };
175 auto df = [=](real m) {
176 real th = std::tanh(4.0 * beta_val * J * m + beta_val * F_val);
177 return 4.0 * beta_val * J * (1.0 - th * th) - 1.0;
178 };
179 auto res = num::newton(f, df, 0.9);
180 return std::max(-1.0, std::min(1.0, res.root));
181 }

193 // Boltzmann acceptance table: boltz_[spin_idx][nbr_idx] = min(1,
194 // exp(-beta*dE)) spin_idx: 0 = s=-1, 1 = s=+1 nbr_idx: (nbr_sum/2 + 2) in
195 // {0..4}
196 void rebuild_boltz() {
197 for (int si = 0; si < 2; ++si) {
198 real s = (si == 0) ? -1.0 : 1.0;
199 for (int ni = 0; ni < 5; ++ni) {
200 real ns = -4.0 + 2.0 * ni;
201 real dE = 2.0 * J * s * ns - 2.0 * F * s;
202 boltz_[si][ni] = num::markov::boltzmann_accept(dE, beta);
203 }
204 }
205 }

207 static int spin_idx(real s) {
208 return s < 0.0 ? 0 : 1;
209 }
210 static int nbr_idx(real ns) {
211 return static_cast<int>(ns * 0.5 + 2.0);
212 }

214 static num::SparseMatrix build_adj() {
215 std::vector<idx> rows_t, cols_t;
216 std::vector<real> vals_t;
217 rows_t.reserve(4 * NN);
218 cols_t.reserve(4 * NN);
219 vals_t.reserve(4 * NN);
220 for (int row = 0; row < N; ++row) {
221 for (int col = 0; col < N; ++col) {
222 int i = row * N + col;
223 int nbrs[4] = {((row - 1 + N) % N) * N + col,
224 ((row + 1) % N) * N + col,
225 row * N + (col - 1 + N) % N,
226 row * N + (col + 1) % N};
227 for (int nb : nbrs) {
228 rows_t.push_back(static_cast<idx>(i));
229 cols_t.push_back(static_cast<idx>(nb));
230 vals_t.push_back(1.0);
231 }
232 }
233 }
234 return num::SparseMatrix::from_triplets(NN, NN, rows_t, cols_t, vals_t);
235 }
