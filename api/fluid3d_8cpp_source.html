61void FluidSolver3D::step() {
62 if (particles_.empty())
63 return;
64
65 for (Particle3D& p : particles_) {
66 p.ax = p.ay = p.az = 0.0f;
67 p.dT_dt = 0.0f;
68 }
69
70 grid_.build(particles_); // O(n + C) counting sort -- always sequential
71
72 const HeatParams3D hp{params_.h,
73 params_.alpha_T,
74 params_.h_conv,
75 params_.mass};
76
77 switch (params_.policy) {
78 case num::Backend::omp:
79 backends::omp::compute_density_pressure(particles_, params_, grid_);
80 backends::omp::compute_forces(particles_, params_, grid_);
81 backends::omp::heat_compute(particles_, bodies_, grid_, hp);
82 backends::omp::body_collisions(particles_, bodies_, params_);
83 backends::omp::integrate(particles_, params_);
84 backends::omp::enforce_boundaries(particles_, params_);
85 backends::omp::update_temp_range(particles_,
86 bodies_,
87 T_min_,
88 T_max_);
89 backends::seq::integrate_bodies(bodies_,
90 params_); // M bodies tiny -- seq
91 break;
92
93 default: // seq: Newton pair traversal
94 backends::seq::compute_density_pressure(particles_, params_, grid_);
95 backends::seq::compute_forces(particles_, params_, grid_);
96 backends::seq::heat_compute(particles_, bodies_, grid_, hp);
97 backends::seq::body_collisions(particles_, bodies_, params_);
98 backends::seq::integrate(particles_, params_);
99 backends::seq::enforce_boundaries(particles_, params_);
100 backends::seq::update_temp_range(particles_,
101 bodies_,
102 T_min_,
103 T_max_);
104 backends::seq::integrate_bodies(bodies_, params_);
105 break;
106 }
107}
