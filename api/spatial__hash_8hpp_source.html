1/// @file spatial_hash.hpp
2/// @brief SPH neighbour search -- now powered by num::CellList2D
3///
4/// Replaces the original chained hash table with the counting-sort cell list
5/// from src/spatial/cell_list.hpp. The public interface is unchanged so that
6/// heat.cpp keeps working without modification.
7///
8/// Improvements over the old chained-list hash:
9///
10/// Old (chained hash table)
11/// build : O(n) -- fill table[] + next[] via random writes
12/// query : O(9 * avg_bucket) -- pointer-chase linked list in each of 9
13/// cells
14///
15/// New (counting sort cell list)
16/// build : O(n + C) -- two sequential passes + one prefix sum
17/// query : O(k) sequential reads -- particles in same cell are contiguous
18/// pairs : iterate_pairs() -- visits each (i,j) pair once (Newton 3rd
19/// law)
20///
21/// The key cache-behaviour difference: the old next_[] array creates a random
22/// walk through the particle array. sorted_[] in CellList2D is laid out so
23/// that all particles in a cell sit next to each other in memory.

32class SpatialHash {
33 public:
34 /// @param cell_size Kernel support radius (2h).
35 /// @param xmin..ymax Simulation domain -- required by CellList2D for the
36 /// flat grid (no hash collisions, zero false positives).
37 SpatialHash(float cell_size, float xmin, float xmax, float ymin, float ymax)
38 : cl_(cell_size, xmin, xmax, ymin, ymax) {}
39
40 /// Rebuild from the particle array. O(n + C).
41 void build(const std::vector<Particle>& particles) {
42 const int n = static_cast<int>(particles.size());
43 cl_.build(
44 [&](int i) {
45 return std::make_pair(particles[i].x, particles[i].y);
46 },
47 n);
48 }
49
50 /// Point query: calls f(int j) for every candidate in the 3x3
51 /// neighbourhood. Caller must verify |r_ij| < cutoff. (Same contract as
52 /// before.)
53 template<typename F>
54 void query(float px, float py, F&& f) const {
55 cl_.query(px, py, std::forward<F>(f));
56 }
57
58 /// Newton's 3rd law pair traversal.
59 /// Calls f(int i, int j) for every unique unordered pair {i,j} whose cells
60 /// are within one cell of each other. Each pair appears exactly once.
61 /// Caller can apply equal-and-opposite contributions to both i and j.
62 template<typename F>
63 void iterate_pairs(F&& f) const {
64 cl_.iterate_pairs(std::forward<F>(f));
65 }
66
67 private:
68 num::CellList2D<float> cl_;
69};
