13void heat_compute(std::vector<Particle>& particles,
14 const std::vector<RigidBody>& bodies,
15 const SpatialHash& grid,
16 const HeatParams& params) {
17 const float h = params.h;
18 const float alpha = params.alpha_T;
19 const float m = params.mass;
20 const float h_conv = params.h_conv;
21 const float supp_sq = 4.0f * h * h;
22 const float eps2 = 0.01f * h * h;
23 const int n = static_cast<int>(particles.size());
24
25 for (int i = 0; i < n; ++i) {
26 Particle& pi = particles[i];
27 float dT = 0.0f;
28
29 grid.query(pi.x, pi.y, [&](int j) {
30 if (j == i)
31 return;
32 const Particle& pj = particles[j];
33 const float rx = pi.x - pj.x, ry = pi.y - pj.y;
34 const float r2 = rx * rx + ry * ry;
35 if (r2 >= supp_sq || r2 < 1e-10f)
36 return;
37 const float r = std::sqrt(r2);
38 const float rij_dot_gradW = K::dW_dr(r, h) * r;
39 dT += 2.0f * alpha * (m / pj.density)
40 * (pi.temperature - pj.temperature) * rij_dot_gradW
41 / (r2 + eps2);
42 });
43
44 for (const RigidBody& body : bodies) {
45 const float dx = pi.x - body.x, dy = pi.y - body.y;
46 const float d = std::sqrt(dx * dx + dy * dy);
47 if (d < body.radius + 2.0f * h) {
48 const float excess = std::max(0.0f, d - body.radius);
49 const float phi = 1.0f - excess / (2.0f * h);
50 dT += h_conv * (body.temperature - pi.temperature) * phi;
51 }
52 }
53
54 pi.dT_dt = dT;
55 }
56}
