13/// @brief Dense matrix exponential via scaling+squaring + Pade [6/6]
14///
15/// Pade [6/6] numerator/denominator coefficients:
16/// c[0]=1, c[1]=1/2, c[2]=5/44, c[3]=1/66,
17/// c[4]=1/792, c[5]=1/15840, c[6]=1/665280
18static Matrix dense_expm(const Matrix& A) {
19 const idx m = A.rows();
20
21 // Pade [6/6] coefficients
22 static constexpr double c[7] = {1.0,
23 0.5,
24 5.0 / 44.0,
25 1.0 / 66.0,
26 1.0 / 792.0,
27 1.0 / 15840.0,
28 1.0 / 665280.0};
29
30 // 1. Compute inf-norm (max absolute row sum)
31 double norm_inf = 0.0;
32 for (idx i = 0; i < m; i++) {
33 double row_sum = 0.0;
34 for (idx j = 0; j < m; j++)
35 row_sum += std::abs(A(i, j));
36 norm_inf = std::max(norm_inf, row_sum);
37 }
38
39 // 2. Scaling: s = max(0, ceil(log2(norm_inf / 0.5)))
40 int s = 0;
41 if (norm_inf > 0.5) {
42 s = (int)std::max(0.0, std::ceil(std::log2(norm_inf / 0.5)));
43 }
44
45 // 3. As = A / 2^s
46 double scale = std::ldexp(1.0, -s); // 1/2^s
47 Matrix As(m, m, 0.0);
48 for (idx i = 0; i < m; i++)
49 for (idx j = 0; j < m; j++)
50 As(i, j) = A(i, j) * scale;
51
52 // 4. Build powers: B = As^2, B2 = As^4, B3 = As^6
53 Matrix B(m, m, 0.0);
54 matmul(As, As, B); // B = As^2
55
56 Matrix B2(m, m, 0.0);
57 matmul(B, B, B2); // B2 = As^4
58
59 Matrix B3(m, m, 0.0);
60 matmul(B2, B, B3); // B3 = As^6
61
62 // 5. V_mat = c[6]*B3 + c[4]*B2 + c[2]*B + c[0]*I (even terms)
63 Matrix V_mat(m, m, 0.0);
64 for (idx i = 0; i < m; i++) {
65 for (idx j = 0; j < m; j++) {
66 V_mat(i, j) = c[6] * B3(i, j) + c[4] * B2(i, j) + c[2] * B(i, j);
67 }
68 V_mat(i, i) += c[0];
69 }
70
71 // 6. W = c[5]*B2 + c[3]*B + c[1]*I (inner odd terms, without As factor)
72 Matrix W(m, m, 0.0);
73 for (idx i = 0; i < m; i++) {
74 for (idx j = 0; j < m; j++) {
75 W(i, j) = c[5] * B2(i, j) + c[3] * B(i, j);
76 }
77 W(i, i) += c[1];
78 }
79
80 // 7. U = As * W (full odd numerator terms)
81 Matrix U(m, m, 0.0);
82 matmul(As, W, U);
83
84 // 8. VpU = V_mat + U, VmU = V_mat - U
85 Matrix VpU(m, m, 0.0);
86 Matrix VmU(m, m, 0.0);
87 for (idx i = 0; i < m; i++) {
88 for (idx j = 0; j < m; j++) {
89 VpU(i, j) = V_mat(i, j) + U(i, j);
90 VmU(i, j) = V_mat(i, j) - U(i, j);
91 }
92 }
93
94 // 9. Solve VmU * E = VpU (Pade approximant: E = (V-U)^{-1}*(V+U))
95 LUResult fac = lu(VmU);
96 Matrix E(m, m, 0.0);
97 lu_solve(fac, VpU, E);
98
99 // 10. Squaring: E = E^(2^s)
100 for (int i = 0; i < s; i++) {
101 Matrix E2(m, m, 0.0);
102 matmul(E, E, E2);
103 E = std::move(E2);
104 }
105
106 return E;
107}

109Vector expv(real t,
110 const MatVecFn& matvec,
111 idx n,
112 const Vector& v,
113 int m_max,
114 real tol) {
115 // Handle zero vector
116 real beta = norm(v);
117 if (beta < 1e-300) {
118 return Vector(n, 0.0);
119 }
120
121 // Krylov basis vectors: V[0..m_max]
122 std::vector<Vector> V;
123 V.reserve(m_max + 1);
124
125 // V[0] = v / beta
126 Vector v0(n);
127 for (idx i = 0; i < n; i++)
128 v0[i] = v[i] / beta;
129 V.push_back(std::move(v0));
130
131 // Upper Hessenberg matrix H: (m_max+1) x m_max
132 Matrix H(m_max + 1, m_max, 0.0);
133
134 int m_actual = m_max;
135
136 // Arnoldi process
137 for (int j = 0; j < m_max; j++) {
138 // w = A * V[j]
139 Vector w(n, 0.0);
140 matvec(V[j], w);
141
142 // Modified Gram-Schmidt orthogonalization
143 for (int i = 0; i <= j; i++) {
144 real h = dot(V[i], w);
145 H(i, j) = h;
146 axpy(-h, V[i], w); // w -= h * V[i]
147 }
148
149 real h_next = norm(w);
150 H(j + 1, j) = h_next;
151
152 // Check for lucky breakdown
153 if (h_next < tol) {
154 m_actual = j + 1;
155 break;
156 }
157
158 // Normalize and store next Krylov vector
159 Vector vj1(n);
160 for (idx i = 0; i < n; i++)
161 vj1[i] = w[i] / h_next;
162 V.push_back(std::move(vj1));
163 }
164
165 // Extract H_m = t * H[0..m_actual-1, 0..m_actual-1]
166 Matrix Hm(m_actual, m_actual, 0.0);
167 for (int i = 0; i < m_actual; i++)
168 for (int j = 0; j < m_actual; j++)
169 Hm(i, j) = t * H(i, j);
170
171 // Compute dense expm of Hm
172 Matrix E = dense_expm(Hm);
173
174 // result = sum_j (beta * E[j,0]) * V[j]
175 Vector result(n, 0.0);
176 for (int j = 0; j < m_actual; j++) {
177 real coeff = beta * E(j, 0);
178 axpy(coeff, V[j], result);
179 }
180
181 return result;
182}
