22ScalarField3D ElectricSolver::joule_heating(const ScalarField3D& sigma,
23 const ScalarField3D& phi) {
24 const num::Grid3D& g = phi.grid();
25 const num::Grid3D& sg = sigma.grid();
26 const int nx = g.nx(), ny = g.ny(), nz = g.nz();
27 const double inv2dx = 1.0 / (2.0 * g.dx());
28
29 ScalarField3D Q(nx, ny, nz, phi.dx(), phi.ox(), phi.oy(), phi.oz());
30 for (int k = 0; k < nz; ++k)
31 for (int j = 0; j < ny; ++j)
32 for (int i = 0; i < nx; ++i) {
33 int ip = std::min(i + 1, nx - 1), im = std::max(i - 1, 0);
34 int jp = std::min(j + 1, ny - 1), jm = std::max(j - 1, 0);
35 int kp = std::min(k + 1, nz - 1), km = std::max(k - 1, 0);
36 double dpdx = (g(ip, j, k) - g(im, j, k)) * inv2dx;
37 double dpdy = (g(i, jp, k) - g(i, jm, k)) * inv2dx;
38 double dpdz = (g(i, j, kp) - g(i, j, km)) * inv2dx;
39 Q.grid().set(i,
40 j,
41 k,
42 sg(i, j, k)
43 * (dpdx * dpdx + dpdy * dpdy + dpdz * dpdz));
44 }
