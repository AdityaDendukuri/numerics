1/// @file src/sim.cpp
2/// @brief Implementation of TDSESolver -- 2-D TDSE via Strang splitting +
3/// Thomas algorithm.
4///
5/// Algorithm chain per step():
6/// 1. kick_V(dt/2) -- diagonal phase multiplication
7/// 2. sweep_x(dt/2) -- Crank-Nicolson in x (Thomas per column fiber)
8/// 3. sweep_y(dt) -- Crank-Nicolson in y (Thomas per row fiber)
9/// 4. sweep_x(dt/2) -- again
10/// 5. kick_V(dt/2) -- again
11///
12/// Eigenstates: num::lanczos on the real N^2xN^2 Hamiltonian (matrix-free
13/// matvec). Bessel zeros: num::brent on J_m(x) for exact CircularWell
14/// eigenvalues.

45void TDSESolver::set_potential(Potential p, double param) {
46 current_potential_ = p;
47 scale(V, 0.0); // zero the potential before rebuilding
48
49 const double cx = L * 0.5, cy = L * 0.5;
50 const double V0 = 5000.0; // hard wall height
51
52 if (p == Potential::Barrier) {
53 // Vertical wall at x = L*0.55, gap centred at y = L/2, width = L*0.12
54 double wall_x = L * 0.55;
55 double gap_cy = cy;
56 double gap_w = L * 0.12;
57 int wall_th = std::max(1, N / 60); // ~4 cells thick
58
59 int ix0 = static_cast<int>((wall_x / L) * N - wall_th);
60 int ix1 = static_cast<int>((wall_x / L) * N + wall_th);
61
62 for (int i = ix0; i <= std::min(ix1, N - 1); ++i) {
63 for (int j = 0; j < N; ++j) {
64 double y = (j + 1.0) * h;
65 if (std::abs(y - gap_cy) > gap_w * 0.5)
66 V[at(i, j)] = V0;
67 }
68 }
69 } else if (p == Potential::DoubleSlit) {
70 double wall_x = L * 0.5;
71 double gap_sep = L * 0.12; // centre-to-centre separation
72 double gap_w = L * 0.06; // each slit width
73 int wall_th = std::max(1, N / 60);
74
75 int ix0 = static_cast<int>((wall_x / L) * N - wall_th);
76 int ix1 = static_cast<int>((wall_x / L) * N + wall_th);
77
78 for (int i = ix0; i <= std::min(ix1, N - 1); ++i) {
79 for (int j = 0; j < N; ++j) {
80 double y = (j + 1.0) * h;
81 bool in_slit1 = std::abs(y - (cy - gap_sep * 0.5))
82 < gap_w * 0.5;
83 bool in_slit2 = std::abs(y - (cy + gap_sep * 0.5))
84 < gap_w * 0.5;
85 if (!in_slit1 && !in_slit2)
86 V[at(i, j)] = V0;
87 }
88 }
89 } else if (p == Potential::Harmonic) {
90 double omega = (param > 0) ? param : 1.5;
91 for (int i = 0; i < N; ++i) {
92 double x = (i + 1.0) * h - cx;
93 for (int j = 0; j < N; ++j) {
94 double y = (j + 1.0) * h - cy;
95 V[at(i, j)] = 0.5 * omega * omega * (x * x + y * y);
96 }
97 }
98 } else if (p == Potential::CircularWell) {
99 double R = (param > 0) ? param : L * 0.4;
100 for (int i = 0; i < N; ++i) {
101 double x = (i + 1.0) * h - cx;
102 for (int j = 0; j < N; ++j) {
103 double y = (j + 1.0) * h - cy;
104 if (std::sqrt(x * x + y * y) > R)
105 V[at(i, j)] = V0;
106 }
107 }
108 }
109 // Free: all zeros already set
110
111 modes_ready = false;
112}

116void TDSESolver::init_gaussian(double x0,
117 double y0,
118 double kx,
119 double ky,
120 double sigma) {
121 for (int i = 0; i < N; ++i) {
122 double x = (i + 1.0) * h;
123 for (int j = 0; j < N; ++j) {
124 double y = (j + 1.0) * h;
125 double dx = x - x0, dy = y - y0;
126 double env = std::exp(-(dx * dx + dy * dy) / (2.0 * sigma * sigma));
127 double phi = kx * dx + ky * dy;
128 psi[at(i, j)] = env * cplx(std::cos(phi), std::sin(phi));
129 }
130 }
131 renormalize();
132}

161void TDSESolver::step() {
162 auto t0 = std::chrono::high_resolution_clock::now();
163
164 kick_V(dt * 0.5);
165 sweep_x(dt * 0.5);
166 sweep_y(dt);
167 sweep_x(dt * 0.5);
168 kick_V(dt * 0.5);
169
170 auto t1 = std::chrono::high_resolution_clock::now();
171 stats.step_ms = std::chrono::duration<double, std::milli>(t1 - t0).count();
172}

181double TDSESolver::compute_energy() const {
182 // <H> = <psi | -1/(2h^2) * Lap + V | psi> * h^2
183 // laplacian_stencil_2d gives lap[k] = (sum of 4 nbrs) - 4*psi[k]
184 // (Dirichlet)
185 CVector lap(psi.size());
186 laplacian_stencil_2d(psi, lap, N);
187
188 const double c = 1.0 / (2.0 * h * h);
189 double E = 0.0;
190 for (int k = 0; k < N * N; ++k) {
191 cplx Hpsi = -c * lap[k] + cplx(V[k]) * psi[k];
192 E += psi[k].real() * Hpsi.real() + psi[k].imag() * Hpsi.imag();
193 }
194 return E * h * h;
195}

202// Eigenstate computation -- Lanczos on the real Hamiltonian
203
204void TDSESolver::compute_modes(int k) {
205 // H is real symmetric -> eigenstates are real. Work entirely in real
206 // arithmetic.
207 const int n2 = N * N;
208 const double c = 1.0 / (2.0 * h * h);
209
210 // Matrix-free Hamiltonian matvec: H = -c * Lap + V (Dirichlet)
211 Vector lap_buf(n2);
212 auto ham_mv = [&](const Vector& x, Vector& y) {
213 laplacian_stencil_2d(x, lap_buf, N);
214 for (int i = 0; i < n2; ++i)
215 y[i] = -c * lap_buf[i] + V[i] * x[i];
216 };
217
218 idx max_steps = std::min(to_idx(5 * k), to_idx(n2));
219 auto res = lanczos(ham_mv, to_idx(n2), k, 1e-8, max_steps);
220
221 modes.clear();
222 int got = static_cast<int>(res.ritz_values.size());
223 modes.reserve(got);
224
225 for (int m = 0; m < got; ++m) {
226 EigenMode em;
227 em.energy = res.ritz_values[m];
228 em.phi.resize(n2);
229
230 double norm_sq = 0;
231 for (int r = 0; r < n2; ++r) {
232 em.phi[r] = res.ritz_vectors(r, m);
233 norm_sq += em.phi[r] * em.phi[r];
234 }
235 double inv = 1.0 / std::sqrt(norm_sq * h * h);
236 for (double& v : em.phi)
237 v *= inv;
238
239 modes.push_back(std::move(em));
240 }
241
242 if (current_potential_ == Potential::CircularWell) {
243 double R = L * 0.4; // default radius (matches set_potential)
244 // Exact energies: E_{m,n} = j_{m,n}^2 / (2R^2)
245 int mode_idx = 0;
246 for (int m = 0; m <= 2 && mode_idx < got; ++m) {
247 double bracket_lo = 1.0;
248 for (int nz = 1; nz <= 3 && mode_idx < got; ++nz) {
249 auto Jm = [m](real x) {
250 return bessel_j(m, x);
251 };
252 double lo = bracket_lo, hi = lo + 0.5;
253 int scan = 0;
254 while (Jm(lo) * Jm(hi) > 0 && scan < 200) {
255 lo = hi;
256 hi += 0.5;
257 ++scan;
258 }
259 if (Jm(lo) * Jm(hi) > 0)
260 break;
261
262 auto res2 = brent(Jm, lo, hi, 1e-10);
263 double E_exact = (res2.root * res2.root) / (2.0 * R * R);
264 bracket_lo = hi + 0.01;
265
266 modes[mode_idx++].exact_energy = E_exact;
267 if (m > 0 && mode_idx < got)
268 modes[mode_idx++].exact_energy = E_exact;
269 }
270 }
271 }
272
273 stats.n_modes = got;
274 modes_ready = true;
275}
