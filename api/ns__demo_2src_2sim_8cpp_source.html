28// Initial condition -- double shear layer
29//
30// Two horizontal shear bands centred at y = 0.25 and y = 0.75.
31// A small vertical perturbation seeds Kelvin-Helmholtz roll-up.
32// The initial field is analytically divergence-free.
33
34void NSSolver::init_shear_layer(real rho, real delta) {
35 const real two_pi = 2.0 * M_PI;
36
37 for (idx i = 0; i < N; ++i) {
38 for (idx j = 0; j < N; ++j) {
39 // u[i,j] lives at (i*h, (j+1/2)*h)
40 real y = (j + 0.5) * h;
41 u[at(i, j)] = (y <= 0.5) ? std::tanh((y - 0.25) / rho)
42 : std::tanh((0.75 - y) / rho);
43
44 // v[i,j] lives at ((i+1/2)*h, j*h)
45 real x = (i + 0.5) * h;
46 v[at(i, j)] = delta * std::sin(two_pi * x);
47 }
48 }
49
50 // Zero pressure (correct for divergence-free initial data)
51 for (idx k = 0; k < N * N; ++k)
52 p[k] = 0.0;
53}

71// Semi-Lagrangian advection
72//
73// For each MAC face, trace a particle backwards in time and interpolate
74// the velocity from the current field. Unconditionally stable for any dt.
75
76void NSSolver::advect() {
77 auto t0 = std::chrono::steady_clock::now();
78
79 // u[i,j] at (i*h, (j+1/2)*h)
80 // Surrounding v-faces for the y-velocity at this point:
81 // v[i-1,j], v[i,j], v[i-1,j+1], v[i,j+1]
82#pragma omp parallel for schedule(static) collapse(2)
83 for (idx i = 0; i < N; ++i) {
84 for (idx j = 0; j < N; ++j) {
85 real uu = u[at(i, j)];
86 real vu = 0.25
87 * (v[at(wm1(i), j)] + v[at(i, j)] + v[at(wm1(i), wp1(j))]
88 + v[at(i, wp1(j))]);
89
90 real xb = i * h - dt * uu;
91 real yb = (j + 0.5) * h - dt * vu;
92 u_star[at(i, j)] = interp_u(xb, yb);
93 }
94 }
95
96 // v[i,j] at ((i+1/2)*h, j*h)
97 // Surrounding u-faces for the x-velocity at this point:
98 // u[i,j-1], u[i+1,j-1], u[i,j], u[i+1,j]
99#pragma omp parallel for schedule(static) collapse(2)
100 for (idx i = 0; i < N; ++i) {
101 for (idx j = 0; j < N; ++j) {
102 real vv = v[at(i, j)];
103 real uv = 0.25
104 * (u[at(i, wm1(j))] + u[at(wp1(i), wm1(j))] + u[at(i, j)]
105 + u[at(wp1(i), j)]);
106
107 real xb = (i + 0.5) * h - dt * uv;
108 real yb = j * h - dt * vv;
109 v_star[at(i, j)] = interp_v(xb, yb);
110 }
111 }
112
113 auto t1 = std::chrono::steady_clock::now();
114 stats.advect_ms =
115 std::chrono::duration<double, std::milli>(t1 - t0).count();
116}

118// Explicit viscosity (forward Euler, stable only when dt <= h^2/(4nu))
119
120void NSSolver::apply_diffusion() {
121 const double c = dt * nu / (h * h);
122 const int n = static_cast<int>(N);
123 pde::diffusion_step_2d(u_star, n, c);
124 pde::diffusion_step_2d(v_star, n, c);
125}

127// RHS: rhs = -div(u*)/dt
128//
129// Discrete divergence on the MAC grid:
130// div(u*)[i,j] = (u*[i+1,j] - u*[i,j]) / h + (v*[i,j+1] - v*[i,j]) / h
131//
132// We form the positive-definite system (-Delta)p = -div(u*)/dt so that CG
133// works.
134
135void NSSolver::build_rhs() {
136 const real scale = -1.0 / (h * dt); // -1/(h*dt) so rhs = -div/dt
137
138#pragma omp parallel for schedule(static) collapse(2)
139 for (idx i = 0; i < N; ++i) {
140 for (idx j = 0; j < N; ++j) {
141 real div_ij = u_star[at(wp1(i), j)] - u_star[at(i, j)]
142 + v_star[at(i, wp1(j))] - v_star[at(i, j)];
143 rhs[at(i, j)] = scale * div_ij;
144 }
145 }
146
147 // Remove mean (periodic Poisson is singular; mean(div) = 0 analytically,
148 // but floating-point errors accumulate -- subtracting keeps CG
149 // well-posed).
150 real sum = 0.0;
151 for (idx k = 0; k < N * N; ++k)
152 sum += rhs[k];
153 real mean = sum / static_cast<real>(N * N);
154 for (idx k = 0; k < N * N; ++k)
155 rhs[k] -= mean;
156}
157
158// Pressure solve: (-Delta)p = rhs via CG
159//
160// (-Delta)p[i,j] = (4p[i,j] - p[i+/-1,j] - p[i,j+/-1]) / h^2
161//
162// The operator is positive semi-definite (null space = constants).
163// We initialise p = previous p (warm start) and subtract the mean after.
164
165void NSSolver::solve_pressure() {
166 auto t0 = std::chrono::steady_clock::now();
167
168 const real inv_h2 = 1.0 / (h * h);
169 const int n = static_cast<int>(N);
170
171 // Matrix-free negative Laplacian via laplacian_stencil_2d_periodic
172 // (boundary-peeled, auto-vectorisable inner loop) then negate and scale.
173 auto neg_lap = [&](const Vector& pin, Vector& out) {
174 laplacian_stencil_2d_periodic(pin, out, n);
175 scale(out, -inv_h2);
176 };
177
178 auto result =
179 num::cg_matfree(neg_lap, rhs, p, /*tol=*/1e-3, /*max_iter=*/100);
180 stats.cg_iters = result.iterations;
181 stats.cg_residual = result.residual;
182
183 // Remove mean from pressure (physics is unchanged; improves stability)
184 real sum = 0.0;
185 for (idx k = 0; k < N * N; ++k)
186 sum += p[k];
187 real mean = sum / static_cast<real>(N * N);
188 for (idx k = 0; k < N * N; ++k)
189 p[k] -= mean;
190
191 auto t1 = std::chrono::steady_clock::now();
192 stats.pressure_ms =
193 std::chrono::duration<double, std::milli>(t1 - t0).count();
194}
195
196// Projection: u = u* - dt*gradp
197//
198// Gradient at MAC face (i,j):
199// (d_p/d_x) at u-face = (p[i,j] - p[i-1,j]) / h
200// (d_p/d_y) at v-face = (p[i,j] - p[i,j-1]) / h
201
202void NSSolver::project() {
203 auto t0 = std::chrono::steady_clock::now();
204
205 const real c = dt / h;
206
207#pragma omp parallel for schedule(static) collapse(2)
208 for (idx i = 0; i < N; ++i) {
209 for (idx j = 0; j < N; ++j) {
210 u[at(i, j)] = u_star[at(i, j)]
211 - c * (p[at(i, j)] - p[at(wm1(i), j)]);
212 v[at(i, j)] = v_star[at(i, j)]
213 - c * (p[at(i, j)] - p[at(i, wm1(j))]);
214 }
215 }
216
217 auto t1 = std::chrono::steady_clock::now();
218 stats.project_ms =
219 std::chrono::duration<double, std::milli>(t1 - t0).count();
220}

224real NSSolver::vorticity(idx i, idx j) const {
225 // omega = d_v/d_x - d_u/d_y at corner (i*h, j*h)
226 // v[i,j] at ((i+1/2)h, j*h), v[i-1,j] at ((i-1/2)h, j*h)
227 // u[i,j] at (i*h, (j+1/2)h), u[i,j-1] at (i*h, (j-1/2)h)
228 real dvdx = (v[at(i, j)] - v[at(wm1(i), j)]) / h;
229 real dudy = (u[at(i, j)] - u[at(i, wm1(j))]) / h;
230 return dvdx - dudy;
231}

233real NSSolver::speed(idx i, idx j) const {
234 // Average faces to cell centre (i+1/2, j+1/2)*h
235 real uc = 0.5 * (u[at(i, j)] + u[at(wp1(i), j)]);
236 real vc = 0.5 * (v[at(i, j)] + v[at(i, wp1(j))]);
237 return std::sqrt(uc * uc + vc * vc);
238}
