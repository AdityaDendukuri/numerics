54static void setup(physics::FluidSolver& solver) {
55 solver.clear();
56 const float h = solver.params().h;
57 const float dx = 0.8f * h;
58
59 for (float x = 0.03f; x < 0.50f; x += dx)
60 for (float y = 0.03f; y < 0.65f; y += dx) {
61 float jx = 0.001f * ((cast<int>(x * 1000) % 7) - 3);
62 float jy = 0.001f * ((cast<int>(y * 1000) % 5) - 2);
63 solver.add_particle(x + jx, y + jy, 0.0f, 0.0f, 20.0f);
64 }
