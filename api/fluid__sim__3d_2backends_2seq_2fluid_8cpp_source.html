15void compute_density_pressure(std::vector<Particle3D>& particles,
16 const FluidParams3D& params,
17 const SpatialHash3D& grid) {
18 const float h = params.h;
19 const float m = params.mass;
20 const float rho0 = params.rho0;
21 const float B = rho0 * params.c0 * params.c0
22 / static_cast<float>(params.gamma);
23 const float supp_sq = 4.0f * h * h;
24
25 const float W0 = K::W(0.0f, h);
26 for (Particle3D& p : particles)
27 p.density = m * W0;
28
29 grid.iterate_pairs([&](int i, int j) {
30 const float rx = particles[i].x - particles[j].x;
31 const float ry = particles[i].y - particles[j].y;
32 const float rz = particles[i].z - particles[j].z;
33 const float r2 = rx * rx + ry * ry + rz * rz;
34 if (r2 >= supp_sq)
35 return;
36 const float w = m * K::W(std::sqrt(r2), h);
37 particles[i].density += w;
38 particles[j].density += w;
39 });
40
41 for (Particle3D& p : particles) {
42 p.density = std::max(p.density, rho0 * 0.1f);
43 p.pressure = std::max(0.0f,
44 B * (num::ipow<7>(p.density / rho0) - 1.0f));
45 }
46}

48void compute_forces(std::vector<Particle3D>& particles,
49 const FluidParams3D& params,
50 const SpatialHash3D& grid) {
51 const float h = params.h;
52 const float m = params.mass;
53 const float mu = params.mu;
54 const float supp_sq = 4.0f * h * h;
55 const float eps2 = 0.01f * h * h;
56
57 for (Particle3D& p : particles) {
58 p.ax = params.gx;
59 p.ay = params.gy;
60 p.az = params.gz;
61 }
62
63 grid.iterate_pairs([&](int i, int j) {
64 Particle3D& pi = particles[i];
65 Particle3D& pj = particles[j];
66 const float rx = pi.x - pj.x, ry = pi.y - pj.y, rz = pi.z - pj.z;
67 const float r2 = rx * rx + ry * ry + rz * rz;
68 if (r2 >= supp_sq || r2 < 1e-10f)
69 return;
70 const float r = std::sqrt(r2);
71
72 auto [gx, gy, gz] = K::Spiky_gradW({rx, ry, rz}, r, h);
73 const float pterm = pi.pressure / (pi.density * pi.density)
74 + pj.pressure / (pj.density * pj.density);
75 const float fpx = -m * pterm * gx, fpy = -m * pterm * gy,
76 fpz = -m * pterm * gz;
77 pi.ax += fpx;
78 pi.ay += fpy;
79 pi.az += fpz;
80 pj.ax -= fpx;
81 pj.ay -= fpy;
82 pj.az -= fpz;
83
84 const float lap = 2.0f * K::Spiky_dW_dr(r, h) * r / (r2 + eps2);
85 const float visc = m * mu / (pi.density * pj.density) * lap;
86 const float fvx = visc * (pi.evx - pj.evx);
87 const float fvy = visc * (pi.evy - pj.evy);
88 const float fvz = visc * (pi.evz - pj.evz);
89 pi.ax += fvx;
90 pi.ay += fvy;
91 pi.az += fvz;
92 pj.ax -= fvx;
93 pj.ay -= fvy;
94 pj.az -= fvz;
95 });
96}

98void body_collisions(std::vector<Particle3D>& particles,
99 const std::vector<RigidBody3D>& bodies,
100 const FluidParams3D& params) {
101 for (Particle3D& p : particles) {
102 for (const RigidBody3D& body : bodies) {
103 const float dx = p.x - body.x, dy = p.y - body.y, dz = p.z - body.z;
104 const float d = std::sqrt(dx * dx + dy * dy + dz * dz);
105 if (d < body.radius && d > 1e-8f) {
106 const float nx = dx / d, ny = dy / d, nz = dz / d;
107 p.x = body.x + body.radius * nx;
108 p.y = body.y + body.radius * ny;
109 p.z = body.z + body.radius * nz;
110 const float vn = p.vx * nx + p.vy * ny + p.vz * nz;
111 if (vn < 0.0f) {
112 const float c = (1.0f + params.restitution) * vn;
113 p.vx -= c * nx;
114 p.vy -= c * ny;
115 p.vz -= c * nz;
116 }
117 }
118 }
119 }

122void integrate(std::vector<Particle3D>& particles,
123 const FluidParams3D& params) {
124 const float dt = params.dt;
125 for (Particle3D& p : particles) {
126 p.vx += p.ax * dt;
127 p.vy += p.ay * dt;
128 p.vz += p.az * dt;
129 p.x += p.vx * dt;
130 p.y += p.vy * dt;
131 p.z += p.vz * dt;
132 p.evx = 0.5f * (p.evx + p.vx);
133 p.evy = 0.5f * (p.evy + p.vy);
134 p.evz = 0.5f * (p.evz + p.vz);
135 p.temperature =
136 std::clamp(p.temperature + p.dT_dt * dt, -100.0f, 300.0f);
137 }

140void enforce_boundaries(std::vector<Particle3D>& particles,
141 const FluidParams3D& params) {
142 const float e = params.restitution;
143 for (Particle3D& p : particles) {
144 if (p.x < params.xmin) {
145 p.x = params.xmin;
146 p.vx = std::abs(p.vx) * e;
147 }
148 if (p.x > params.xmax) {
149 p.x = params.xmax;
150 p.vx = -std::abs(p.vx) * e;
151 }
152 if (p.y < params.ymin) {
153 p.y = params.ymin;
154 p.vy = std::abs(p.vy) * e;
155 }
156 if (p.y > params.ymax) {
157 p.y = params.ymax;
158 p.vy = -std::abs(p.vy) * e;
159 }
160 if (p.z < params.zmin) {
161 p.z = params.zmin;
162 p.vz = std::abs(p.vz) * e;

188void integrate_bodies(std::vector<RigidBody3D>& bodies,
189 const FluidParams3D& params) {
190 const float dt = params.dt, e = params.restitution;
191 for (RigidBody3D& b : bodies) {
192 if (b.fixed)
193 continue;
194 b.vx += params.gx * dt;
195 b.vy += params.gy * dt;
196 b.vz += params.gz * dt;
197 b.x += b.vx * dt;
198 b.y += b.vy * dt;
199 b.z += b.vz * dt;
200 const float r = b.radius;
201 if (b.x - r < params.xmin) {
202 b.x = params.xmin + r;
203 b.vx = std::abs(b.vx) * e;
204 }
205 if (b.x + r > params.xmax) {
206 b.x = params.xmax - r;
207 b.vx = -std::abs(b.vx) * e;
208 }
209 if (b.y - r < params.ymin) {
210 b.y = params.ymin + r;
211 b.vy = std::abs(b.vy) * e;
212 }
213 if (b.y + r > params.ymax) {
214 b.y = params.ymax - r;
215 b.vy = -std::abs(b.vy) * e;
216 }
217 if (b.z - r < params.zmin) {
218 b.z = params.zmin + r;
219 b.vz = std::abs(b.vz) * e;
