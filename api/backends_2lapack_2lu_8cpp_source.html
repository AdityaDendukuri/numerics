18LUResult lu(const Matrix& A) {
19#if defined(NUMERICS_HAS_LAPACK)
20 const idx n = A.rows();
21 LUResult f;
22 f.LU = A;
23 f.piv.resize(n);
24 f.singular = false;
25
26 std::vector<lapack_int> ipiv(n);
27 int info = LAPACKE_dgetrf(LAPACK_ROW_MAJOR,
28 static_cast<lapack_int>(n),
29 static_cast<lapack_int>(n),
30 f.LU.data(),
31 static_cast<lapack_int>(n),
32 ipiv.data());
33 if (info < 0)
34 throw std::runtime_error("lu (lapack): dgetrf argument error, info="
35 + std::to_string(info));
36 if (info > 0)
37 f.singular = true;
38
39 // Convert 1-based LAPACK IPIV to 0-based sequential swap targets
40 for (idx k = 0; k < n; ++k)
41 f.piv[k] = static_cast<idx>(ipiv[k] - 1);
42
43 return f;
44#else
45 return seq::lu(A);
46#endif
47}
