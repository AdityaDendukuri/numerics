28float ScalarField3D::sample(float x, float y, float z) const {
29 const float gx = (x - ox_) / dx();
30 const float gy = (y - oy_) / dx();
31 const float gz = (z - oz_) / dx();
32
33 if (gx < 0 || gx >= nx() - 1 || gy < 0 || gy >= ny() - 1 || gz < 0
34 || gz >= nz() - 1)
35 return 0.0f;
36
37 const int i0 = static_cast<int>(gx);
38 const int j0 = static_cast<int>(gy);
39 const int k0 = static_cast<int>(gz);
40 const float tx = gx - i0, ty = gy - j0, tz = gz - k0;
41
42 auto v = [&](int di, int dj, int dk) {
43 return static_cast<float>(grid_(i0 + di, j0 + dj, k0 + dk));
44 };
45 return (1 - tz)
46 * ((1 - ty) * ((1 - tx) * v(0, 0, 0) + tx * v(1, 0, 0))
47 + ty * ((1 - tx) * v(0, 1, 0) + tx * v(1, 1, 0)))
48 + tz
49 * ((1 - ty) * ((1 - tx) * v(0, 0, 1) + tx * v(1, 0, 1))
50 + ty * ((1 - tx) * v(0, 1, 1) + tx * v(1, 1, 1)));
51}

87SolverResult FieldSolver::solve_poisson(ScalarField3D& phi,
88 const ScalarField3D& source,
89 double tol,
90 int max_iter) {
91 const Grid3D& g = phi.grid();
92 const int nx = g.nx(), ny = g.ny(), nz = g.nz();
93 const double inv_dx2 = 1.0 / (g.dx() * g.dx());
94 const int N = nx * ny * nz;
95
96 auto flat = [&](int i, int j, int k) -> idx {
97 return static_cast<idx>(k * ny * nx + j * nx + i);
98 };
99
100 Vector b(N, 0.0);
101 for (int k = 1; k < nz - 1; ++k) {
102 for (int j = 1; j < ny - 1; ++j) {
103 for (int i = 1; i < nx - 1; ++i) {
104 b[flat(i, j, k)] = -source.grid()(i, j, k);
105 }
106 }
107 }
108
109 Vector xv = phi.grid().to_vector();
110
111 auto matvec = [&](const Vector& v, Vector& Av) {
112 neg_laplacian_3d(v, Av, nx, ny, nz, inv_dx2);
113 };
114 auto result = cg_matfree(matvec, b, xv, tol, static_cast<idx>(max_iter));
115 phi.grid().from_vector(xv);
116 return result;
117}

159VectorField3D MagneticSolver::current_density(const ScalarField3D& sigma,
160 const ScalarField3D& phi) {
161 VectorField3D J = FieldSolver::gradient(phi);
162 const Grid3D& sg = sigma.grid();
163 const int nx = sg.nx(), ny = sg.ny(), nz = sg.nz();
164 for (int k = 0; k < nz; ++k)
165 for (int j = 0; j < ny; ++j)
166 for (int i = 0; i < nx; ++i) {
167 const double neg_s = -sg(i, j, k);
168 J.x.grid()(i, j, k) *= neg_s;
169 J.y.grid()(i, j, k) *= neg_s;
170 J.z.grid()(i, j, k) *= neg_s;
171 }
172 return J;
