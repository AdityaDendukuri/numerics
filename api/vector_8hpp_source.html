193/// @brief Non-owning view of a flat Vector as an array of 2D points.
194///
195/// Many physics simulations pack 2D positions or velocities into a flat
196/// Vector with stride 2: element i has x = v[2i], y = v[2i+1].
197/// Vec2View provides named accessors that eliminate the raw index arithmetic
198/// without copying or reshaping the data.
199///
200/// Example:
201/// @code
202/// num::Vector q(2 * N); // flat storage: [x0,y0, x1,y1, ...]
203/// num::Vec2View pos{q};
204/// pos.x(i) = 1.0; // writes q[2*i]
205/// pos.y(i) = 2.0; // writes q[2*i+1]
206/// idx n = pos.size(); // = N
207/// @endcode
208struct Vec2View {
209 Vector& v;
210
211 /// Number of 2D points (= v.size() / 2).
212 idx size() const noexcept {
213 return v.size() / 2;
214 }
215
216 real& x(idx i) noexcept {
217 return v[2 * i];
218 }
219 real x(idx i) const noexcept {
220 return v[2 * i];
221 }
222 real& y(idx i) noexcept {
223 return v[2 * i + 1];
224 }
225 real y(idx i) const noexcept {
226 return v[2 * i + 1];
227 }
228};
229
230/// @brief Read-only variant of Vec2View for const Vectors.
231struct Vec2ConstView {

233
234 idx size() const noexcept {

236 }
237 real x(idx i) const noexcept {

242 }
