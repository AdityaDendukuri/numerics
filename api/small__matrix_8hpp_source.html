1/// @file small_matrix.hpp
2/// @brief Constexpr fixed-size stack-allocated matrix and Givens rotation.
3///
4/// These types complement the heap-allocated Matrix / Vector for small inner
5/// kernels where heap traffic is undesirable:
6///
7/// SmallVec<N> -- constexpr dense vector backed by std::array<real,N>
8/// SmallMatrix<M,N> -- constexpr row-major matrix backed by
9/// std::array<real,M*N> GivensRotation -- constexpr (c,s) pair; apply()
10/// / apply_t() are branchless
11///
12/// Because these are literal types every operation can be evaluated at
13/// compile time (C++17: arithmetic only; C++20+: also sqrt/hypot via
14/// constexpr `<cmath>`). At runtime the compiler places them entirely in
15/// registers for small M, N.
16///
17/// Typical uses:
18/// - GMRES: Hessenberg column updates via GivensRotation::from + apply
19/// - Jacobi SVD / eig: 2x2 eigensystem kernel as SmallMatrix<2,2>
20/// - Register-tile accumulation: SmallMatrix<4,4> replaces 16 named doubles
21/// - Unit tests: known-answer matmuls computed at compile time

165/// @brief Givens plane rotation: G = [c, s; -s, c].
166///
167/// Constructed so that G * [a; b]^T = [r; 0] with r = hypot(a, b).
168///
169/// @code
170/// auto g = GivensRotation::from(h_jj, h_jj1); // zero H[j,j+1] in GMRES
171/// g.apply(g_j, g_j1); // rotate RHS vector too
172/// @endcode
173struct GivensRotation {
174 real c, s;
175
176 /// Compute (c,s) such that G*[a;b] = [r;0], r = hypot(a,b).
177 /// Stable for any (a,b) including b==0.
178 static constexpr GivensRotation from(real a, real b) noexcept {
179 if (b == real(0))
180 return {real(1), real(0)};
181 // Use the standard stable form. std::hypot is constexpr in GCC/Clang
182 // as a built-in extension even in C++17; becomes standard in C++23.
183 real r = std::sqrt(a * a + b * b);
184 return {a / r, b / r};
185 }
186
187 /// Apply G from the left: [x; y] <- G * [x; y].
188 constexpr void apply(real& x, real& y) const noexcept {
189 real tmp = c * x + s * y;
190 y = -s * x + c * y;
191 x = tmp;
192 }
193
194 /// Apply G^T (inverse rotation): [x; y] <- G^T * [x; y].
195 constexpr void apply_t(real& x, real& y) const noexcept {
196 real tmp = c * x - s * y;
197 y = s * x + c * y;
198 x = tmp;
199 }
200
201 /// Return the 2x2 rotation matrix.
202 constexpr SmallMatrix<2, 2> as_matrix() const noexcept {
203 return SmallMatrix<2, 2>{{c, s, -s, c}};
204 }
205};
