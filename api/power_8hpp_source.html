33/// Normalise v in-place; returns the old norm.
34inline real normalise(Vector& v) {
35 real nrm = 0;
36 for (idx i = 0; i < v.size(); ++i)
37 nrm += v[i] * v[i];
38 nrm = std::sqrt(nrm);
39 if (nrm > 1e-300)
40 for (idx i = 0; i < v.size(); ++i)
41 v[i] /= nrm;
