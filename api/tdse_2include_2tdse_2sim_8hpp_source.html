1/// @file include/tdse/sim.hpp
2/// @brief 2-D Time-Dependent Schrodinger Equation solver
3///
4/// Algorithm: Strang operator splitting
5/// e^{-iHdt} ~= e^{-iVdt/2} * e^{-iTx*dt/2} * e^{-iTy*dt} * e^{-iTx*dt/2} *
6/// e^{-iVdt/2}
7///
8/// Each kinetic sub-step uses Crank-Nicolson with a complex tridiagonal solve
9/// (Thomas algorithm). Potential kick is a diagonal phase multiplication: psi
10/// *= exp(-i*V*tau).
11///
12/// Eigenstate computation uses num::lanczos on the real Hamiltonian matrix (H
13/// is real symmetric for real V, so eigenstates are real).
14///
15/// Bessel function zeros (for CircularWell exact eigenvalues) are found via
16/// num::brent. Norm/energy observables use num::gauss_legendre on radial
17/// marginals.
18///
19/// Grid: NxN interior points, domain [0,L]x[0,L], Dirichlet BCs (psi=0 on
20/// boundary). Storage: row-major idx = i*N + j, i = row (x), j = col (y).

37enum class Potential {
38 Free, ///< V = 0 (free particle)
39 Barrier, ///< Single vertical barrier with one gap -- tunnelling
40 DoubleSlit, ///< Double-slit wall -- interference
41 Harmonic, ///< V = 1/2*omega^2*r^2 (coherent state dynamics)
42 CircularWell, ///< V = 0 inside radius R, V = V0 outside (Bessel
43 ///< eigenstates)
44};

46inline const char* potential_name(Potential p) {
47 switch (p) {
48 case Potential::Free:
49 return "Free particle";
50 case Potential::Barrier:
51 return "Single barrier";
52 case Potential::DoubleSlit:
53 return "Double slit";
54 case Potential::Harmonic:
55 return "Harmonic oscillator";
56 case Potential::CircularWell:
