1/// @file backends/omp/fluid.cpp
2/// @brief OpenMP SPH backend -- parallel for over particles
3///
4/// Thread safety contract (same principle as src/backends/omp/matrix.cpp):
5/// Each parallel section has a unique "owner thread" for writes.
6/// Thread i reads particles[j].* fields that were finalised in a prior
7/// sequential phase and are not modified during the current section.
8/// Thread i writes only to particles[i].* -> no data races without atomics.
9///
10/// Why per-particle query instead of Newton pairs:
11/// iterate_pairs writes to both particles[i] and particles[j].
12/// Parallelising over pairs would need atomic float adds -- expensive and
13/// unreadable. Per-particle query keeps each thread isolated to index i.
14/// Thread-level speedup offsets the 2x more pair evaluations for n >~ 500.
15///
16/// Falls back to seq backend when NUMERICS_HAS_OMP is not defined,
17/// identical to the pattern in src/backends/omp/vector.cpp.

36void compute_density_pressure(std::vector<Particle>& particles,
37 const FluidParams& params,
38 const SpatialHash& grid) {
39#ifdef NUMERICS_HAS_OMP
40 const float h = params.h;
41 const float m = params.mass;
42 const float rho0 = params.rho0;
43 const float B = rho0 * params.c0 * params.c0
44 / static_cast<float>(params.gamma);
45 const float supp_sq = 4.0f * h * h;
46 const int n = static_cast<int>(particles.size());
47
48 #pragma omp parallel for schedule(static)
49 for (int i = 0; i < n; ++i) {
50 Particle& pi = particles[i];
51 float rho = 0.0f;
52 grid.query(pi.x, pi.y, [&](int j) {
53 const float rx = pi.x - particles[j].x;
54 const float ry = pi.y - particles[j].y;
55 const float r2 = rx * rx + ry * ry;
56 if (r2 < supp_sq)
57 rho += m * K::W(std::sqrt(r2), h);
58 });
59 pi.density = std::max(rho, rho0 * 0.1f);
60 pi.pressure = std::max(0.0f,
61 B * (num::ipow<7>(pi.density / rho0) - 1.0f));
62 }
63#else
64 seq::compute_density_pressure(particles, params, grid);
65#endif
66}

68void compute_forces(std::vector<Particle>& particles,
69 const FluidParams& params,
70 const SpatialHash& grid) {
71#ifdef NUMERICS_HAS_OMP
72 const float h = params.h;
73 const float m = params.mass;
74 const float mu = params.mu;
75 const float supp_sq = 4.0f * h * h;
76 const float eps2 = 0.01f * h * h;
77 const int n = static_cast<int>(particles.size());
78
79 #pragma omp parallel for schedule(static)
80 for (int i = 0; i < n; ++i) {
81 const Particle& pi = particles[i];
82 float ax = params.gx, ay = params.gy;
83 grid.query(pi.x, pi.y, [&](int j) {
84 if (j == i)
85 return;
86 const Particle& pj = particles[j];
87 const float rx = pi.x - pj.x, ry = pi.y - pj.y;
88 const float r2 = rx * rx + ry * ry;
89 if (r2 >= supp_sq || r2 < 1e-10f)
90 return;
91 const float r = std::sqrt(r2);
92
93 auto [gx, gy] = K::Spiky_gradW({rx, ry}, r, h);
94 const float pterm = pi.pressure / (pi.density * pi.density)
95 + pj.pressure / (pj.density * pj.density);
96 ax -= m * pterm * gx;
97 ay -= m * pterm * gy;
98
99 const float lap = 2.0f * K::Spiky_dW_dr(r, h) * r / (r2 + eps2);
100 const float visc = m * mu / (pi.density * pj.density) * lap;
101 ax += visc * (pi.evx - pj.evx);
102 ay += visc * (pi.evy - pj.evy);
103 });
104 particles[i].ax = ax;
105 particles[i].ay = ay;
106 }
107#else
108 seq::compute_forces(particles, params, grid);
109#endif
110}

112void body_collisions(std::vector<Particle>& particles,
113 const std::vector<RigidBody>& bodies,
114 const FluidParams& params) {
115#ifdef NUMERICS_HAS_OMP
116 const int n = static_cast<int>(particles.size());
117 #pragma omp parallel for schedule(static)
118 for (int i = 0; i < n; ++i) {
119 Particle& p = particles[i];
120 for (const RigidBody& body : bodies) {
121 const float dx = p.x - body.x, dy = p.y - body.y;
122 const float d = std::sqrt(dx * dx + dy * dy);
123 if (d < body.radius && d > 1e-8f) {
124 const float nx = dx / d, ny = dy / d;
125 p.x = body.x + body.radius * nx;
126 p.y = body.y + body.radius * ny;
127 const float vn = p.vx * nx + p.vy * ny;
128 if (vn < 0.0f) {
129 p.vx -= (1.0f + params.restitution) * vn * nx;
130 p.vy -= (1.0f + params.restitution) * vn * ny;
131 }
132 }
133 }
134 }
135#else
136 seq::body_collisions(particles, bodies, params);
137#endif
138}

140void integrate(std::vector<Particle>& particles, const FluidParams& params) {
141#ifdef NUMERICS_HAS_OMP
142 const float dt = params.dt;
143 const int n = static_cast<int>(particles.size());
144 #pragma omp parallel for schedule(static)
145 for (int i = 0; i < n; ++i) {
146 Particle& p = particles[i];
147 p.vx += p.ax * dt;
148 p.vy += p.ay * dt;
149 p.x += p.vx * dt;
150 p.y += p.vy * dt;
151 p.evx = 0.5f * (p.evx + p.vx);
152 p.evy = 0.5f * (p.evy + p.vy);
153 p.temperature =
154 std::clamp(p.temperature + p.dT_dt * dt, -100.0f, 300.0f);
155 }
156#else
157 seq::integrate(particles, params);
158#endif
159}

161void enforce_boundaries(std::vector<Particle>& particles,
162 const FluidParams& params) {
163#ifdef NUMERICS_HAS_OMP
164 const float e = params.restitution;
165 const int n = static_cast<int>(particles.size());
166 #pragma omp parallel for schedule(static)
167 for (int i = 0; i < n; ++i) {
168 Particle& p = particles[i];
169 if (p.x < params.xmin) {
170 p.x = params.xmin;
171 p.vx = std::abs(p.vx) * e;
172 }
173 if (p.x > params.xmax) {
174 p.x = params.xmax;
175 p.vx = -std::abs(p.vx) * e;
176 }
177 if (p.y < params.ymin) {
178 p.y = params.ymin;
179 p.vy = std::abs(p.vy) * e;
180 }
181 if (p.y > params.ymax) {
182 p.y = params.ymax;
183 p.vy = -std::abs(p.vy) * e;
184 }
185 }
186#else
187 seq::enforce_boundaries(particles, params);
188#endif
189}

191void update_temp_range(const std::vector<Particle>& particles,
192 const std::vector<RigidBody>& bodies,
193 float& T_min,
194 float& T_max) {
195#ifdef NUMERICS_HAS_OMP
196 if (particles.empty())
197 return;
198 const int n = static_cast<int>(particles.size());
199 float lo = FLT_MAX, hi = -FLT_MAX;
200 #pragma omp parallel for reduction(min : lo) reduction(max : hi) \
201 schedule(static)
202 for (int i = 0; i < n; ++i) {
203 lo = std::min(lo, particles[i].temperature);
204 hi = std::max(hi, particles[i].temperature);
205 }
206 T_min = lo;
207 T_max = hi;
208 for (const RigidBody& b : bodies) { // M bodies -- sequential, M is small
209 T_min = std::min(T_min, b.temperature);
210 T_max = std::max(T_max, b.temperature);
211 }
212#else
213 seq::update_temp_range(particles, bodies, T_min, T_max);
214#endif
215}
