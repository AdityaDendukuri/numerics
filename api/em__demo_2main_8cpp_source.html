1/// @file apps/em_demo/main.cpp
2/// @brief Electromagnetic field demo -- aluminum rod + permanent magnet.
3///
4/// Physics solved once at startup (CG Poisson solvers):
5/// Electric: div(sigma * grad(phi)) = 0 -> J = sigma * (-grad phi)
6/// Magnetic: Laplacian(A) = -mu0 * J -> B = curl(A)
7/// Magnet dipole field added analytically each frame.
8/// Camera orbits the scene; ESC quits.

42static bool inside_rod(int i, int k) {
43 int di = i - ROD_CX, dk = k - ROD_CZ;
44 return di * di + dk * dk <= ROD_R * ROD_R;
45}
46static int flat(int i, int j, int k) {
47 return k * NY * NX + j * NX + i;
48}

50// Dipole field (m = M0 * y_hat)
51static void dipole_B(float wx,
52 float wy,
53 float wz,
54 float mx,
55 float my,
56 float mz,
57 float& bx,
58 float& by,
59 float& bz) {
60 float rx = wx - mx, ry = wy - my, rz = wz - mz;
61 float r2 = rx * rx + ry * ry + rz * rz;
62 if (r2 < 1e-5f) {
63 bx = by = bz = 0.0f;
64 return;
65 }
66 float r5 = r2 * r2 * sqrtf(r2);
67 float mdr = MAG_M0 * ry;
68 constexpr float C = 1e-7f;
69 bx = C * 3.0f * mdr * rx / r5;
70 by = C * MAG_M0 * (3.0f * ry * ry - r2) / r5;
71 bz = C * 3.0f * mdr * rz / r5;
72}
