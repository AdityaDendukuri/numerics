1/// @file eigen/lanczos.cpp
2/// @brief Lanczos algorithm with full reorthogonalisation
3///
4/// The k-step Lanczos recurrence:
5///
6/// beta_0 = 0, v_0 = 0
7/// v_1 = random unit vector
8/// for j = 1..k:
9/// w = A*v_j
10/// alpha_j = v_j^T * w
11/// w = w - alpha_j*v_j - beta_{j-1}*v_{j-1}
12/// [reorthogonalise w against all previous v_1..v_j]
13/// beta_j = ||w||
14/// v_{j+1} = w / beta_j
15///
16/// This builds the symmetric tridiagonal T_k = V_k^T * A * V_k with diagonal
17/// alpha and off-diagonal beta. The Ritz values are the eigenvalues of T_k and
18/// the Ritz vectors are V_k * (eigenvectors of T_k).
19///
20/// T_k is diagonalised with eig_sym (Jacobi), which is O(k^3) -- cheap for k
21/// << n.
22

31LanczosResult lanczos(MatVecFn matvec,
32 idx n,
33 idx k,
34 real tol,
35 idx max_steps,
36 Backend /*backend*/) {
37 if (k == 0 || k > n)
38 throw std::invalid_argument("lanczos: k must satisfy 0 < k <= n");
39
40 if (max_steps == 0)
41 max_steps = std::min(3 * k, n);
42 max_steps = std::min(max_steps, n);
43
44 // Lanczos basis: columns of V (n x max_steps)
45 // Store as a flat array for efficient column access
46 Matrix V(n, max_steps, 0.0);
47
48 // alpha (diagonal) and beta (off-diagonal) of the tridiagonal T
49 Vector alpha(max_steps, 0.0);
50 Vector beta(max_steps, 0.0); // beta[j] = beta_{j+1}, beta[0] unused
51
52 // Initialise v_1 = e_1 (deterministic, avoids Rng dependency)
53 for (idx i = 0; i < n; ++i)
54 V(i, 0) = (i == 0) ? 1.0 : 0.0;
55
56 idx steps = 0;
57
58 for (idx j = 0; j < max_steps; ++j) {
59 // Extract current Lanczos vector v_j from column j of V
60 Vector vj(n);
61 for (idx i = 0; i < n; ++i)
62 vj[i] = V(i, j);
63
64 // w = A * vj
65 Vector w(n, 0.0);
66 matvec(vj, w);
67
68 // alpha_j = v_j^T * w
69 real a = 0;
70 for (idx i = 0; i < n; ++i)
71 a += vj[i] * w[i];
72 alpha[j] = a;
73
74 // w = w - alpha_j*v_j - beta_{j-1}*v_{j-1}
75 for (idx i = 0; i < n; ++i)
76 w[i] -= a * vj[i];
77 if (j > 0) {
78 for (idx i = 0; i < n; ++i)
79 w[i] -= beta[j - 1] * V(i, j - 1);
80 }
81
82 // Full reorthogonalisation (modified Gram-Schmidt against all previous)
83 for (idx l = 0; l <= j; ++l) {
84 real proj = 0;
85 for (idx i = 0; i < n; ++i)
86 proj += V(i, l) * w[i];
87 for (idx i = 0; i < n; ++i)
88 w[i] -= proj * V(i, l);
89 }
90
91 // beta_j = ||w||
92 real b = 0;
93 for (idx i = 0; i < n; ++i)
94 b += w[i] * w[i];
95 b = std::sqrt(b);
96
97 ++steps;
98
99 // Invariant subspace found -- can't continue
100 if (b < 1e-12)
101 break;
102
103 beta[j] = b;
104
105 // Store v_{j+1} if room remains
106 if (j + 1 < max_steps) {
107 for (idx i = 0; i < n; ++i)
108 V(i, j + 1) = w[i] / b;
109 }
110 }
111
112 // Build steps x steps tridiagonal T from (alpha, beta) and diagonalise it.
113 // Using all 'steps' Lanczos vectors (not just k) gives far better Ritz
114 // approximations.
115 idx m = steps;
116 Matrix T(m, m, 0.0);
117 for (idx j = 0; j < m; ++j) {
118 T(j, j) = alpha[j];
119 if (j + 1 < m) {
120 T(j, j + 1) = beta[j];
121 T(j + 1, j) = beta[j];
122 }
123 }
124
125 EigenResult teig = eig_sym(T, tol * 1e-2); // tighter tol for inner solve
126 // teig.values are in ascending order; the top-k are at indices m-k .. m-1
127
128 idx nret = std::min(k, m); // how many Ritz pairs to return
129
130 // Ritz vectors for the top-nret eigenvalues: V_m * eigvecs_of_T[:,
131 // m-nret..m-1]
132 Matrix ritz_vecs(n, nret, 0.0);
133 for (idx i = 0; i < nret; ++i) {
134 idx ti = m - nret + i; // column in teig (ascending order; largest last)
135 for (idx j = 0; j < m; ++j) {
136 real coeff = teig.vectors(j, ti);
137 for (idx r = 0; r < n; ++r)
138 ritz_vecs(r, i) += coeff * V(r, j);
139 }
140 }
141
142 // Slice the top-nret eigenvalues (ascending)
143 Vector ritz_vals(nret);
144 for (idx i = 0; i < nret; ++i)
145 ritz_vals[i] = teig.values[m - nret + i];
146
147 // Check convergence of each Ritz pair: ||A*u - lambda*u||
148 bool all_converged = true;
149 for (idx i = 0; i < nret; ++i) {
150 Vector u(n);
151 for (idx r = 0; r < n; ++r)
152 u[r] = ritz_vecs(r, i);
153
154 Vector Au(n, 0.0);
155 matvec(u, Au);
156
157 real res = 0;
158 real lam = ritz_vals[i];
159 for (idx r = 0; r < n; ++r) {
160 real d = Au[r] - lam * u[r];
161 res += d * d;
162 }
163 if (std::sqrt(res) > tol) {
164 all_converged = false;
165 break;
166 }
167 }
168
169 return {ritz_vals, ritz_vecs, steps, all_converged};
170}
